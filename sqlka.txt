CREATE DATABASE IF NOT EXISTS loja_pws;
USE loja_pws;

CREATE TABLE loja (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    nif VARCHAR(9) NOT NULL,
    morada VARCHAR(255) NOT NULL,
    telefone VARCHAR(15),
    email VARCHAR(100)
);

CREATE TABLE produtos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    referencia VARCHAR(50) UNIQUE NOT NULL,
    descricao VARCHAR(255) NOT NULL,
    preco_unitario DECIMAL(10,2) NOT NULL
);

CREATE TABLE faturas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    numero_fatura VARCHAR(50) UNIQUE NOT NULL,
    data_emissao DATETIME DEFAULT CURRENT_TIMESTAMP,
    nome_cliente VARCHAR(100) NOT NULL,
    morada_cliente VARCHAR(255),
    nif_cliente VARCHAR(9) NOT NULL,
    valor_total DECIMAL(10,2) DEFAULT 0.00
);

CREATE TABLE linhas_fatura (
    id INT AUTO_INCREMENT PRIMARY KEY,
    fatura_id INT NOT NULL,
    produto_id INT NOT NULL,
    quantidade INT NOT NULL,
    subtotal DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (fatura_id) REFERENCES faturas(id) ON DELETE CASCADE,
    FOREIGN KEY (produto_id) REFERENCES produtos(id)
);
